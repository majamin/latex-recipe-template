\recipeTitle{Chocolate Chip Cookies}
\recipeDescription{A classic soft and chewy cookie loaded with rich chocolate chips.}
\recipeInfo{Prep: 15 min \quad Bake: 10 min \quad Yield: 24 cookies}

\begin{Recipe}

	\begin{RecipeIngredients}
		\ingred{1 cup salted butter softened}
		\ingred{1 cup granulated sugar}
		\ingred{1 cup light brown sugar packed}
		\ingred{2 teaspoons pure vanilla extract}
		\ingred{2 large eggs}
		\ingred{3 cups all-purpose flour}
		\ingred{1 teaspoon baking soda}
		\ingred{½ teaspoon baking powder}
		\ingred{1 teaspoon sea salt}
		\ingred{2 cups chocolate chips (12 oz)}
	\end{RecipeIngredients}
	\begin{RecipeDirections}
		\direct{Preheat oven to 375$^\circ$ F. Line three baking sheets with parchment paper and set aside.}
		\direct{In a medium bowl mix flour, baking soda, baking powder and salt. Set aside.}
		\direct{Cream together butter and sugars until combined.}
		\direct{Beat in eggs and vanilla until light (about 1 minute).}
		\direct{Mix in the dry ingredients until combined.}
		\direct{Add chocolate chips and mix well.}
		\direct{Roll 2-3 Tablespoons (depending on how large you like your cookies) of dough at a time into balls and place them evenly spaced on your prepared cookie sheets.}
		\direct{Bake in preheated oven for approximately 8-10 minutes. Take them out when they are just barely starting to turn brown.}
		\direct{Let them sit on the baking pan for 5 minutes before removing to cooling rack.}
	\end{RecipeDirections}

\end{Recipe}

\begin{RecipeNote}
	This dough freezes well. Bake straight from frozen, adding 2–3 minutes to baking time.
\end{RecipeNote}
